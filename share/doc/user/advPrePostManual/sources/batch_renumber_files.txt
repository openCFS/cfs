#!/bin/sh
 
INPUT_FILES="manfred-001.bmp stefan_1981.png simon6_28.jpg andreas-00099.tif"
 
for FILE in $INPUT_FILES; do
    # Remove other parts of file name. Just keep file number
    FILE_NUM=`echo $FILE | sed 's/\([a-zA-Z0-9_]*\)\([_\-]\)\([0-9]*\)\(.[a-zA-Z0-9]*\)$/\3/g'`;
 
    # Remove leading zeros
    FILE_NUM=`echo $FILE_NUM |sed 's/0*//'`;
 
    # Print as four character number with leading zeros
    FILE_NUM=`printf %4.d $FILE_NUM | sed -e 's/ /0/g'`;
 
    # Generate and echo new file name
    NEW_FILE_NAME="out_$FILE_NUM.png";
    echo "$NEW_FILE_NAME (from $FILE)";
done
