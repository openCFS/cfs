% ===============================
%  XML Listing Definitions
% ===============================

% The following definitions should ease the pain of 
% handling xml elements and attributes

% background color for attribute / element listings
\definecolor{darkgrey}{rgb}{0.8,0.8,0.8}

% single xml <element> within text
\newcommand\xel[1]{\textcolor[rgb]{0.00,0.50,0.00}{
\tt{\textless#1\textgreater}}}

% single closing </element> within txt
\newcommand\xelclose[1]{\textcolor[rgb]{0.00,0.50,0.00}{
\tt{\textless/#1\textgreater}}}

% single empty <element/> within txt
\newcommand\xelempty[1]{\textcolor[rgb]{0.00,0.50,0.00}{
\tt{\textless #1/\textgreater}}}

% single xml element value within the text
\newcommand\xelv[1]{{\tt{#1}}}

% single xml attribute within the text
\newcommand\xat[1]{\textcolor[rgb]{0.49,0.56,0.16}{\tt{#1}}}

% single xml attribute value within the text
\newcommand\xatv[1]{\textcolor[rgb]{0.73,0.13,0.13}{\tt{#1}}}

% generic xml value (no specfic value, just the name / meaning
% of the parameter)
\newcommand\xparam[1]{\textit{#1}}

% definition list for attributes in the form
% The first parameter names the description of the attribute
% | Attribute | Values | Description |
\newenvironment{xatlist}[1]
{\begin{center} \newcommand{\attrDesc}{#1}
\begin{tabular}{lp{0.3\textwidth}p{0.5\textwidth}} \toprule \rowcolor{darkgrey}
Attribute & Values & Description \\ \midrule} {\\\bottomrule
\end{tabular}\\[0.2ex] \attrDesc \end{center}}



% definition list for element values in the form
\newenvironment{xellist}[1]
{\begin{center} \newcommand{\xelDesc}{#1} \begin{tabular}{llp{5cm}}
\toprule \rowcolor{darkgrey} Element & Values & Description \\ \midrule}
{\end{tabular} \\[0.2ex] \xelDesc\end{center}}

\newcommand{\xml}[1]{#1}
