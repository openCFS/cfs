# Depending on USE_SCRIPTING_TCL/PYTHON, we have to include the
# tcl/python headers, as they get referenced already in the
# tcl/py-messenger header
IF(USE_TCL)
  INCLUDE_DIRECTORIES(${TCL_INCLUDE_PATH})
ENDIF(USE_TCL)

IF(USE_PYTHON)
  INCLUDE_DIRECTORIES(${PYTHON_INCLUDE_PATH})
ENDIF(USE_PYTHON)

IF(USE_HDF5)
  INCLUDE_DIRECTORIES(${HDF5_INCLUDE_DIR})
ENDIF(USE_HDF5)

SET(DEFINEFILES_SRCS definefiles.cc)

ADD_LIBRARY(definefiles STATIC ${DEFINEFILES_SRCS})

IF(USE_TCL)
  SET(TARGET_LL ${TARGET_LL} tclmess scripting)
ENDIF(USE_TCL)

IF(USE_PYTHON)
  SET(TARGET_LL ${TARGET_LL} pythonmess scripting)
ENDIF(USE_PYTHON)

IF(USE_MESH)
  SET(TARGET_LL ${TARGET_LL} siminmesh)
  SET(TARGET_LL ${TARGET_LL} internalmesh)
ENDIF(USE_MESH)

IF(USE_GMV)
  IF(USE_GMV_INPUT)
    SET(TARGET_LL ${TARGET_LL} simingmv)
  ENDIF(USE_GMV_INPUT)

  IF(USE_GMV_OUTPUT)
    SET(TARGET_LL ${TARGET_LL} simoutgmv)
  ENDIF(USE_GMV_OUTPUT)
ENDIF(USE_GMV)

IF(USE_GMSH)
  SET(TARGET_LL ${TARGET_LL} siminoutgmsh)
ENDIF(USE_GMSH)

IF(USE_GIDPOST)
  SET(TARGET_LL ${TARGET_LL} simoutgid)
ENDIF(USE_GIDPOST)


IF(USE_UNV)
  SET(TARGET_LL ${TARGET_LL} simoutunv)
ENDIF(USE_UNV)

IF(USE_HDF5)
  SET(TARGET_LL ${TARGET_LL} siminouthdf5)
  SET(TARGET_LL ${TARGET_LL} simoutxdmf)
ENDIF(USE_HDF5)

IF(USE_ANSYSRST)
  SET(TARGET_LL ${TARGET_LL} simoutansysrst)
ENDIF(USE_ANSYSRST)

IF(USE_VTK)
  SET(TARGET_LL ${TARGET_LL} simoutvtk)
ENDIF(USE_VTK)

SET(TARGET_LL ${TARGET_LL} simoutinfo)

SET(TARGET_LL ${TARGET_LL} datainout)

TARGET_LINK_LIBRARIES(definefiles
  ${TARGET_LL}
  )
