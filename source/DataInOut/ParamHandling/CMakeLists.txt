  ParamNode.cc
  XMLMaterialHandler.cc
  XmlReader.cc
)

    ${PARAMH_SRCS}

ENDIF(USE_XERCES)

  INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR})

  SET(PARAMH_SRCS
    ${PARAMH_SRCS}
    LibXml2.cc)
ENDIF(USE_LIBXML2)

  materials

ENDIF(USE_XERCES)

  LIST(APPEND TARGET_LL ${LIBXML2_LIBRARY})
ENDIF(USE_LIBXML2)  


TARGET_LINK_LIBRARIES(paramh ${TARGET_LL})
