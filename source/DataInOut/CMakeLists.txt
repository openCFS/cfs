IF(USE_XERCES)
  INCLUDE_DIRECTORIES(${XERCES_INCLUDE_DIR})
ENDIF(USE_XERCES)

IF(CFS_BLAS_LAPACK STREQUAL "MKL")
  SET_SOURCE_FILES_PROPERTIES(ProgramOptions.cc
    PROPERTIES COMPILE_FLAGS "-I${MKL_INCLUDE_DIR} -fno-strict-aliasing")
ENDIF(CFS_BLAS_LAPACK STREQUAL "MKL")

SET(DATAINOUT_SRCS 
  ColoredConsole.cc
  PostProc.cc
  ProgramOptions.cc
  ResultCache.cc
  ResultHandler.cc
  SimOutput.cc
)

ADD_LIBRARY(datainout STATIC ${DATAINOUT_SRCS})

SET(TARGET_LL
  simouttext
  simoutinfo
  )

IF(NOT MINGW)
  LIST(APPEND TARGET_LL
    simoutstreaming
    )
ENDIF(NOT MINGW)

LIST(APPEND TARGET_LL
  ${BOOST_PROGRAM_OPTIONS_LIB}
  ${BOOST_FILESYSTEM_LIB}
  ${BOOST_SYSTEM_LIB}
  ${BZIP2_LIBRARY}
  )

IF(USE_XERCES)
  LIST(APPEND TARGET_LL
    ${XERCES_LIBRARY}
    )
ENDIF(USE_XERCES)

IF(USE_LAPACK)
  LIST(APPEND TARGET_LL
    ${LAPACK_LIBRARY}
    ${BLAS_LIBRARY}
    ${CFS_FORTRAN_LIBS}
    )
ENDIF(USE_LAPACK)


TARGET_LINK_LIBRARIES(datainout ${TARGET_LL})
