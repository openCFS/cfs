INCLUDE_DIRECTORIES(
  ${HDF5_INCLUDE_DIR}
  ${CMAKE_CURRENT_BINARY_DIR}
  ${CMAKE_CURRENT_SOURCE_DIR})

IF(USE_IPOPT)
  INCLUDE_DIRECTORIES(${IPOPT_INCLUDE_DIR})
  LINK_DIRECTORIES(${IPOPT_LINK_DIR})
ENDIF(USE_IPOPT)

# List of source codes for current target.
SET(CFSTOOL_SRCS
  cfstool.cc
  ParamsInit.cc
  fft.cc
)

# Main CFS++ Tool executable target.
ADD_EXECUTABLE(cfstoolbin ${CFSTOOL_SRCS})

SET(TARGET_LL
  cfsgeneral
  datainout  
  domain
  elements
  matvec
  paramh
  pde
  utils
  
  ${BOOST_SERIALIZATION_LIB}
  ${BOOST_FILESYSTEM_LIB}
  ${BOOST_SYSTEM_LIB}
)

#check if cfstool is on for fftw
OPTION(CFSTOOL_FFTW "Build support for the FFTW library to perform FFTs and IFFTs of hdf5 files" ON)
MARK_AS_ADVANCED(CFSTOOL_FFTW)

IF(USE_GMV)
  IF(USE_GMV_INPUT)
    SET(TARGET_LL ${TARGET_LL} simingmv)
  ENDIF(USE_GMV_INPUT)

  IF(USE_GMV_OUTPUT)
    SET(TARGET_LL ${TARGET_LL} simoutgmv)
  ENDIF(USE_GMV_OUTPUT)
ENDIF(USE_GMV)

IF(USE_GIDPOST)
  SET(TARGET_LL ${TARGET_LL} simoutgid)
ENDIF(USE_GIDPOST)

IF(USE_MESH)
  SET(TARGET_LL ${TARGET_LL} siminmesh)
ENDIF(USE_MESH)

IF(USE_UNV)
  SET(TARGET_LL ${TARGET_LL} simoutunv)
ENDIF(USE_UNV)

IF(USE_HDF5)
  INCLUDE_DIRECTORIES(${GIDPOST_INCLUDE_DIR})
  SET(TARGET_LL ${TARGET_LL} siminouthdf5 simoutxdmf)
ENDIF(USE_HDF5)

IF(CFSTOOL_FFTW)
  INCLUDE_DIRECTORIES(${FFTW_INCLUDE_DIR}) 
  SET(TARGET_LL ${TARGET_LL} fftw3)
ENDIF(CFSTOOL_FFTW)

IF(USE_ANSYSRST)
  SET(TARGET_LL ${TARGET_LL} simoutansysrst)
ENDIF(USE_ANSYSRST)

IF(USE_GMSH)
  SET(TARGET_LL ${TARGET_LL} siminoutgmsh)
ENDIF(USE_GMSH)

IF(USE_TCL)
  SET(TARGET_LL ${TARGET_LL} tclmess scripting)
ENDIF(USE_TCL)

IF(USE_PYTHON)
  SET(TARGET_LL ${TARGET_LL} pythonmess scripting)
ENDIF(USE_PYTHON)



SET(TARGET_LL ${TARGET_LL} 
  ${CFS_FORTRAN_DYNRT_LIBS}
)




TARGET_LINK_LIBRARIES(cfstoolbin ${TARGET_LL})

# Make sure CFS Tool gets linked with system libgcc.so instead of ANSYS libgcc.so.
IF(USE_ANSYSRST)
  SET_TARGET_PROPERTIES(cfstoolbin PROPERTIES LINK_FLAGS -Wl,-rpath,/${LIB_SUFFIX})
ENDIF(USE_ANSYSRST)
