OPTION(CPLREADER_ANSYS "" ON)  
OPTION(CPLREADER_CFX "" ON)  
OPTION(CPLREADER_FASTEST "" ON)  
OPTION(CPLREADER_OPENFOAM "" ON)  
OPTION(CPLREADER_MPCCI "" OFF)

MARK_AS_ADVANCED(CPLREADER_ANSYS)
MARK_AS_ADVANCED(CPLREADER_CFX)
MARK_AS_ADVANCED(CPLREADER_FASTEST)
MARK_AS_ADVANCED(CPLREADER_OPENFOAM)
MARK_AS_ADVANCED(CPLREADER_MPCCI)

INCLUDE(cplreader_options.cmake)
  
INCLUDE_DIRECTORIES(
  ${HDF5_INCLUDE_DIR}
  ${CMAKE_CURRENT_BINARY_DIR}
  ${CMAKE_CURRENT_SOURCE_DIR})

IF(CPLREADER_ANSYS)
  SUBDIRS(FileReaders/ANSYS)
ENDIF(CPLREADER_ANSYS)

IF(CPLREADER_CFX)
  SUBDIRS(FileReaders/CFX FileReaders/CFXexport)
ENDIF(CPLREADER_CFX)

IF(CPLREADER_FASTEST)
  SUBDIRS(FileReaders/FASTEST)
ENDIF(CPLREADER_FASTEST)

IF(CPLREADER_OPENFOAM)
  SUBDIRS(FileReaders/OPENFOAM)
ENDIF(CPLREADER_OPENFOAM)

SUBDIRS(FileReaders/GENGRIDS)

SUBDIRS(OutputWriters/hdf5)
IF(CPLREADER_MPCCI AND USE_MPCCI)
  SUBDIRS(OutputWriters/MpCCI)
ENDIF(CPLREADER_MPCCI AND USE_MPCCI)

SUBDIRS(Documentation)

SET(CPLREADER_SRCS 
  CouplingHandler.cc
  cplreader.cc
  FileReader.cc
  OutputWriter.cc
  ${CMAKE_CURRENT_BINARY_DIR}/ParamsInit.cc
  integlib/elemIntegr.cc
#  integlib/integr.cc
)

ADD_EXECUTABLE(cplreader ${CPLREADER_SRCS})

SET(TARGET_LL ${TARGET_LL}
  siminouthdf5
  forms
  elements
  matrix
  utils
  cfsgeneral
)

IF(CPLREADER_ANSYS)
  SET(TARGET_LL ${TARGET_LL} cplreader_ansys)
ENDIF(CPLREADER_ANSYS)

IF(CPLREADER_CFX)
  SET(TARGET_LL
  ${TARGET_LL}
  cplreader_cfx
  cplreader_cfxexport)
ENDIF(CPLREADER_CFX)

IF(CPLREADER_FASTEST)
  SET(TARGET_LL ${TARGET_LL} cplreader_fastest)
ENDIF(CPLREADER_FASTEST)

IF(CPLREADER_OPENFOAM)
  SET(TARGET_LL ${TARGET_LL} cplreader_openfoam)
ENDIF(CPLREADER_OPENFOAM)

SET(TARGET_LL ${TARGET_LL}
  cplreader_gengrids
  cplreader_docu
  cplreader_hdf5
)

IF(CPLREADER_MPCCI AND USE_MPCCI)
  SET(TARGET_LL ${TARGET_LL}
    cplreader_mpcci
    )
ENDIF(CPLREADER_MPCCI AND USE_MPCCI)

SET(TARGET_LL ${TARGET_LL}
  ${BOOST_PROGRAM_OPTIONS_LIB}
  # DIRTY, DIRTYYYY HACK!
  datainout
  simouttext
  datainout
  )
  
TARGET_LINK_LIBRARIES(cplreader ${TARGET_LL})

IF(CPLREADER_CFX)
  SET_TARGET_PROPERTIES(cplreader PROPERTIES
    LINK_FLAGS -Wl,--wrap,__ctype_b
    LINKER_LANGUAGE CXX)
ENDIF(CPLREADER_CFX)
