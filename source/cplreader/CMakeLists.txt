OPTION(CPLREADER_ANSYS "" ON)  
IF(NOT CFS_DISTRO STREQUAL "MACOSX")
  OPTION(CPLREADER_CFX "" ON)  
ENDIF(NOT CFS_DISTRO STREQUAL "MACOSX")
OPTION(CPLREADER_FASTEST "" ON)  
OPTION(CPLREADER_FIELDVIEW "" ON)  
OPTION(CPLREADER_OPENFOAM "" OFF)  
OPTION(CPLREADER_ENSIGHT "" OFF)
OPTION(CPLREADER_FLUENT "" OFF)
OPTION(CPLREADER_CGNS "" OFF)  
OPTION(CPLREADER_MPCCI "" OFF)
# If documentation is to be rebuilt set this to 1
OPTION(CPLREADER_REBUILD_DOCU "" OFF)

MARK_AS_ADVANCED(CPLREADER_ANSYS)
MARK_AS_ADVANCED(CPLREADER_CFX)
MARK_AS_ADVANCED(CPLREADER_FASTEST)
MARK_AS_ADVANCED(CPLREADER_FIELDVIEW)
MARK_AS_ADVANCED(CPLREADER_OPENFOAM)
MARK_AS_ADVANCED(CPLREADER_ENSIGHT)
MARK_AS_ADVANCED(CPLREADER_FLUENT)
MARK_AS_ADVANCED(CPLREADER_CGNS)
MARK_AS_ADVANCED(CPLREADER_MPCCI)
MARK_AS_ADVANCED(CPLREADER_REBUILD_DOCU "" OFF)

INCLUDE(cplreader_options.cmake)
  
INCLUDE_DIRECTORIES(
  ${HDF5_INCLUDE_DIR}
  ${CMAKE_CURRENT_BINARY_DIR}
  ${CMAKE_CURRENT_SOURCE_DIR})

SUBDIRS(FileReaders/CFS++)
IF(CPLREADER_ANSYS)
  SUBDIRS(FileReaders/ANSYS)
ENDIF(CPLREADER_ANSYS)

IF(CPLREADER_CFX)
  SUBDIRS(FileReaders/CFX FileReaders/CFXexport)
ENDIF(CPLREADER_CFX)

IF(CPLREADER_FASTEST)
  SUBDIRS(FileReaders/FASTEST)
ENDIF(CPLREADER_FASTEST)

IF(CPLREADER_FIELDVIEW)
  SUBDIRS(FileReaders/FieldView)
ENDIF(CPLREADER_FIELDVIEW)

IF(CPLREADER_OPENFOAM OR
   CPLREADER_ENSIGHT OR
   CPLREADER_FLUENT)
  SUBDIRS(FileReaders/VTKBased)
ENDIF(CPLREADER_OPENFOAM OR
  CPLREADER_ENSIGHT OR
  CPLREADER_FLUENT)

IF(CPLREADER_CGNS)
  SUBDIRS(FileReaders/CGNS)
ENDIF(CPLREADER_CGNS)

SUBDIRS(FileReaders/GENGRIDS)

SUBDIRS(OutputWriters/hdf5)

SUBDIRS(Documentation)

SET(CPLREADER_SRCS 
  CouplingHandler.cc
  cplreader.cc
  FileReader.cc
  OutputWriter.cc
  ${CMAKE_CURRENT_BINARY_DIR}/ParamsInit.cc
  integlib/elemIntegr.cc
#  integlib/integr.cc
)

ADD_EXECUTABLE(cplreader ${CPLREADER_SRCS})

SET(TARGET_LL ${TARGET_LL}
  siminouthdf5
  elements
  mathparser
  matvec
  utils
  cfsgeneral
  cplreader_cfs
)

IF(CPLREADER_ANSYS)
  SET(TARGET_LL ${TARGET_LL} cplreader_ansys)
ENDIF(CPLREADER_ANSYS)

IF(CPLREADER_CFX)
  SET(TARGET_LL
  ${TARGET_LL}
  ${CFX_LIBIO}
  cplreader_cfx
  ${CFX_LIBIO}
  cplreader_cfx
  cplreader_cfxexport)
ENDIF(CPLREADER_CFX)

IF(CPLREADER_FASTEST)
  SET(TARGET_LL ${TARGET_LL} cplreader_fastest)
ENDIF(CPLREADER_FASTEST)

IF(CPLREADER_FIELDVIEW)
  SET(TARGET_LL ${TARGET_LL} cplreader_fieldview)
ENDIF(CPLREADER_FIELDVIEW)

IF(CPLREADER_OPENFOAM)
  SET(TARGET_LL ${TARGET_LL} cplreader_openfoam)
ENDIF(CPLREADER_OPENFOAM)

IF(CPLREADER_ENSIGHT)
  SET(TARGET_LL ${TARGET_LL} cplreader_ensight)
ENDIF(CPLREADER_ENSIGHT)

IF(CPLREADER_FLUENT)
  SET(TARGET_LL ${TARGET_LL} cplreader_fluent)
ENDIF(CPLREADER_FLUENT)

IF(CPLREADER_CGNS)
  SET(TARGET_LL ${TARGET_LL} cplreader_cgns)
ENDIF(CPLREADER_CGNS)

SET(TARGET_LL ${TARGET_LL}
  cplreader_gengrids
  cplreader_docu
  cplreader_hdf5
)

SET(TARGET_LL ${TARGET_LL}
  ${BOOST_PROGRAM_OPTIONS_LIB}
  # DIRTY, DIRTYYYY HACK!
  datainout
  simouttext
  datainout
  )

SET(TARGET_LL ${TARGET_LL}
  ${CFS_FORTRAN_DYNRT_LIBS}
  )

TARGET_LINK_LIBRARIES(cplreader ${TARGET_LL})

IF(CPLREADER_CFX)
  SET_TARGET_PROPERTIES(cplreader PROPERTIES
    LINK_FLAGS -Wl,--wrap,__ctype_b
    LINKER_LANGUAGE CXX)
ENDIF(CPLREADER_CFX)
