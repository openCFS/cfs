OPTION(CPLREADER_ANSYS "" ON)  
OPTION(CPLREADER_CFX "" ON)  
OPTION(CPLREADER_FASTEST "" ON)  
OPTION(CPLREADER_OPENFOAM "" ON)  

MARK_AS_ADVANCED(CPLREADER_ANSYS)
MARK_AS_ADVANCED(CPLREADER_CFX)
MARK_AS_ADVANCED(CPLREADER_FASTEST)
MARK_AS_ADVANCED(CPLREADER_OPENFOAM)
  
INCLUDE_DIRECTORIES(
  ${MPCCI_INCLUDE_DIR}
  ${HDF5_INCLUDE_DIR})

IF(CPLREADER_ANSYS)
  SUBDIRS(FileReaders/ANSYS)
ENDIF(CPLREADER_ANSYS)

IF(CPLREADER_CFX)
  SUBDIRS(FileReaders/CFX FileReaders/CFXexport)
ENDIF(CPLREADER_CFX)

IF(CPLREADER_FASTEST)
  SUBDIRS(FileReaders/FASTEST)
ENDIF(CPLREADER_FASTEST)

IF(CPLREADER_OPENFOAM)
  SUBDIRS(FileReaders/OPENFOAM)
ENDIF(CPLREADER_OPENFOAM)

SET(CPLREADER_SRCS 
  cplreader.cc
  FileReader.cc
  MpCCIexch.cc
  ParamsInit.cc
  integlib/elemIntegr.cc
  integlib/integr.cc
)

ADD_EXECUTABLE(cplreader ${CPLREADER_SRCS})

SET(TARGET_LL ${TARGET_LL}
  siminouthdf5
  forms
  elements
  matrix
  utils
  cfsgeneral
)

IF(CPLREADER_ANSYS)
  SET(TARGET_LL ${TARGET_LL} cplreader_ansys)
ENDIF(CPLREADER_ANSYS)

IF(CPLREADER_CFX)
  SET(TARGET_LL
  ${TARGET_LL}
  cplreader_cfx
  cplreader_cfxexport)
ENDIF(CPLREADER_CFX)

IF(CPLREADER_FASTEST)
  SET(TARGET_LL ${TARGET_LL} cplreader_fastest)
ENDIF(CPLREADER_FASTEST)

IF(CPLREADER_OPENFOAM)
  SET(TARGET_LL ${TARGET_LL} cplreader_openfoam)
ENDIF(CPLREADER_OPENFOAM)

SET(TARGET_LL ${TARGET_LL}
  ${BOOST_PROGRAM_OPTIONS_LIB}
  # DIRTY, DIRTYYYY HACK!
  datainout
  simouttext
  datainout
)

TARGET_LINK_LIBRARIES(cplreader ${TARGET_LL})

IF(CPLREADER_CFX)
  SET_TARGET_PROPERTIES(cplreader PROPERTIES
    LINK_FLAGS -Wl,--wrap,__ctype_b
    LINKER_LANGUAGE CXX)
ENDIF(CPLREADER_CFX)
